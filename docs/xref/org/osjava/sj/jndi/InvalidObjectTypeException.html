43  /***
44   * An exception to indicate that the object that was being bound to a context 
45   * is not a valid type for the context.
46   * 
47   * @author Robert M. Zigweid
48   * @version $Rev: 1978 $ $Date: 2005-08-29 18:30:33 -0700 (Mon, 29 Aug 2005) $
49   * @since  0.9.2 XXX: Henri confirm this
50   */
51  public class InvalidObjectTypeException extends NamingException {
52  
53      /***
54       * Create a new InvalidObjectTypeException with the given explanation.
55       * 
56       * @param explanation An explanation of why the exception was thrown.
57       */
58      public InvalidObjectTypeException(String explanation) {
59          super(explanation);
60      }
61  
62      /***
63       * Create a new InvalidObjectTypeException with the default explanation.
64       */
65      public InvalidObjectTypeException() {
66          super("Objects of this type are not supported by the context.");
67      }

69  }
